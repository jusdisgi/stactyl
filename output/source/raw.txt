# =============================================================================
# stactyl flat plate — PER-COLUMN COMB (BastardKB-style). Model confirmed with
# Hunter 2026-06-22; see ../stactyl/CLAUDE.md "PCB approach".
# -----------------------------------------------------------------------------
# NOT a grid of facets. Each finger column is ONE continuous full-width strip
# (keys at row pitch; the strip flexes along the scoop in the gaps BETWEEN
# switches — no neck-down between rows). Adjacent columns are joined by a single
# serpentine neck (alternating top/bottom). Everything terminates at a rigid
# controller root (XIAO nRF52840 Plus + power/reset/battery) that folds into the
# under-keywell pocket. Outboard pinky necks off the pinky strip; the 2-key
# thumb cluster is its own strip on a serpentine tail off the root.
#
# Wiring: ~5x4 matrix (cols C0..C4, rows R0..R3) + one SOD-323 diode/key.
# Column lines run UP each strip (never cross a neck); row + RGB + 5V/GND are
# the only nets crossing the necks. RGB: one serial SK6805-1515 (EC15) chain,
# LED in each switch cutout, snaking the comb (see chain order in the PCB
# scaffold at the bottom).
#
# WHAT'S RUNNABLE HERE: the points (matrix layout: 5 finger columns + outboard
# pinky + 2 thumbs). The staggers/splays are a STARTING POINT — dial them in
# ergogen against cad/stactyl-cosmos-expert.ts + a bent test print (the necks
# absorb the residual, so this does NOT need an exact arc-length unfold).
# The OUTLINE (strip rects + necks + root) and the PCB section (footprints +
# LED chain) are scaffolded in comments at the bottom — iterate those in ergogen
# on Hunter's machine (per workspace memory, generators run there, not sandbox).
# =============================================================================

units:
  # PG1316S preset — mirrored from ../xiphos / ../LightFury.
  kx: 16
  ky: 16
  ks: 17    # column spacing (PG1316S, 17 across a row)
  kp: 17    # row spacing (key-to-key down a column strip)
  kcow: 16  # cutout width  (= stock cap; caps bottom out by design)
  kcoh: 16  # cutout height
  keycw: 16
  keych: 16

  # Trace / via (from ../LightFury)
  signal_trace_width: 0.2
  pwr_trace_width: 0.25
  gnd_trace_width: 0.25
  via_size: 0.5    # JLCPCB no-extra-charge min 0.56 (KiCad default 0.8)
  via_drill: 0.3   # JLCPCB no-extra-charge min 0.3

  # Finger column stagger — flat-plate Y offset per column. STARTING values;
  # tune so the bent comb lands on its keywell posts (necks take up the rest).
  stagger_finger_pinky: -4
  stagger_finger_ring: 3
  stagger_finger_middle: 5
  stagger_finger_index: 0
  stagger_finger_inner: -2

  # Finger column splay (deg). Keep ~0 for flat strips — the keywell's row
  # curvature is absorbed by the inter-column necks, not in-plane splay.
  splay_finger_pinky: 0
  splay_finger_ring: 0
  splay_finger_middle: 0
  splay_finger_index: 0
  splay_finger_inner: 0

points:
  zones:
    # ---- Finger field: 5 columns x 3 rows. One column = one comb strip. ----
    # Columns laid left->right (right hand): inner, index, middle, ring, pinky.
    # column_net runs UP each strip; row_net crosses the necks.
    finger:
      key:
        padding: kp
        spread: ks
      columns:
        inner:
          key:
            column_net: C4
            stagger: stagger_finger_inner
            splay: splay_finger_inner
        index:
          key:
            column_net: C3
            stagger: stagger_finger_index
            splay: splay_finger_index
        middle:
          key:
            column_net: C2
            stagger: stagger_finger_middle
            splay: splay_finger_middle
        ring:
          key:
            column_net: C1
            stagger: stagger_finger_ring
            splay: splay_finger_ring
        pinky:
          key:
            column_net: C0
            stagger: stagger_finger_pinky
            splay: splay_finger_pinky
      rows:
        bottom:
          row_net: R2
        home:
          row_net: R1
        top:
          row_net: R0

    # ---- Outboard pinky: 1 key, necks off the pinky strip (xiphos/TOTEM style).
    # Shares the pinky column wire (C0), its own matrix row R3. Rotated outward;
    # 1u (custom rounded/tri cap + clearance handled in CAD). Anchor/rotation
    # are a STARTING POINT — dial in ergogen.
    pinky_out:
      anchor:
        ref: finger_pinky_home
        shift: [ks, 0]
      key:
        column_net: C0
        row_net: R3
        rotate: -15
      columns:
        out:
      rows:
        mid:

    # ---- Thumb cluster: 2 keys on their own strip / serpentine tail off the
    # root. Curved 2-key (per the brief). column_net C2/C3, row R3. Anchor +
    # splay are a STARTING POINT — set the real angle from the Cosmos thumb
    # transforms (cad/stactyl-cosmos-expert.ts, leftThumbsPlane/rightThumbsPlane).
    thumb:
      anchor:
        ref: finger_index_bottom
        shift: [4, -ky - 6]
        rotate: -20
      key:
        padding: ks + 1
        spread: ks + 1
      columns:
        home:
          key:
            column_net: C2
            row_net: R3
        far:
          key:
            column_net: C3
            row_net: R3
            splay: -18
      rows:
        fan:

# OUTLINE — scaffold (build/iterate in ergogen on Hunter's machine).
# The comb is NOT a single rounded outline. Build it as:
#   - one rounded-rect strip per finger column wrapping its 3 keys (full width,
#     ~ a switch body + margin), 0.6 mm FR-4;
#   - a small serpentine neck between each adjacent column pair, alternating the
#     top vs bottom edge down the hand (see CLAUDE.md);
#   - a stub off the pinky strip for the outboard pinky;
#   - a rigid root rectangle at the proximal edge (XIAO Plus + power + battery),
#     with a serpentine tail to the thumb strip and a neck up to the inner col;
#   - subtract switch/LED cutouts as the footprints define them.
# Necks should be drawn as explicit polygons, not auto-hull, so their width +
# serpentine shape are controlled for bend tolerance.
# =============================================================================

#   - per_key_leds: SK6805-1515 / EC15 footprint (TO BUILD via kicad2ergogen;

#   - mcu: XIAO nRF52840 Plus on the rigid root (confirm the footprint from
#     ../ergogen-footprints xiao_ble_plus_breakout_holes / util_xiao_ble_plus_*).
#   - power_switch_smd_side, reset_switch_smd_side, battery connector on the root.
#
# RGB CHAIN ORDER (single serial line; assign led_prev/led_next in points to
# match once wiring): root DIN -> thumb_far -> thumb_home -> inner(bottom->home
# ->top) -> index(top->home->bottom) -> middle(bottom->home->top) -> ring(top->
